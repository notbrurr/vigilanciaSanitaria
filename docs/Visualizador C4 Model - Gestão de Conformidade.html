    Component(auth, "Camada de Autenticação", "Segurança / Criptografia", "Algoritmos de hash e proteção de dados sensíveis (LGPD).")
    Component(model, "Camada de Model", "Mapeamento Relacional", "Entidades transacionais e estruturas de junção.")
    Component(logger, "Logger Interceptor", "Auditoria", "Escuta mutações de dados para registro de rastreabilidade.")
}
ContainerDb(db, "SQLite", "Database")

Rel(template, view, "Envia requisições de usuário")
Rel(view, auth, "Solicita validação de acesso")
Rel(view, model, "Solicita processamento de negócio")
Rel(auth, model, "Repassa dados criptografados")
Rel(model, logger, "Notifica alteração de dados")
Rel(logger, db, "Grava log de auditoria", "SQL")
Rel(model, db, "Persiste dados de negócio", "SQL")
@enduml
